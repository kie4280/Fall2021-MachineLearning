# Linear regression

from sklearn.preprocessing import PolynomialFeatures
from sklearn.linear_model import LinearRegression
import numpy as np
import matplotlib.pyplot as plt

# data generation
Xs = np.random.uniform(-1.5, 1, size=1000)
noise = np.random.normal(0, 0.5, size=Xs.shape)
Ys = 3 * Xs**3 + 2 * Xs**2 - 3 * Xs + 1 + noise

f = plt.figure(figsize=(15, 10), dpi=80)
_ = plt.scatter(Xs, Ys)
res = []

for k in range(1, 5):
    pof = PolynomialFeatures(degree=k)

    trans = pof.fit_transform(np.expand_dims(Xs, axis=1))
    # print(trans)

    lr = LinearRegression()
    lr.fit(trans, Ys)

    coef = lr.coef_.T
    coef[0] = lr.intercept_

    line = np.expand_dims(np.linspace(-1.5, 1, 1000), axis=1)
    trans = pof.fit_transform(line)
    # print(coef)
    y = np.matmul(trans, coef)
    lab = "${:.2f}".format(coef[0])
    for i in range(1, k+1):
        c = coef[i]
        lab += ("+" if c >= 0 else "") + "{:.2f} \\times ".format(c) + ("x_i" if i==1 else "x^{}_i".format(i))
        print(c, i)
    # print(y)
    lab += "$"
    _ = plt.plot(line, y, label=lab)
plt.legend()
